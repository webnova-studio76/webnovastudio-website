How it works

This project uses Object-Oriented Programming to represent books and the library itself. We use Dictionaries to track inventory and Loops to provide a continuous user menu.

Uses Input/Output for user interaction and While Loops for the menu.

Module 3

Module 4 & 5

Uses Classes for structure and Functions for actions like lending.

Source Code

class Book:
    def __init__(self, title, author):
        self.title = title
        self.author = author
        self.is_available = True

class Library:
    def __init__(self):
        self.books = []

    def add_book(self, book):
        self.books.append(book)

    def show_books(self):
        for book in self.books:
            status = "Available" if book.is_available else "Borrowed"
            print(f"- {book.title} by {book.author} [{status}]")

# Main Logic
my_library = Library()
my_library.add_book(Book("Python Basics", "Code Craft"))
my_library.add_book(Book("Web Magic", "Jane Doe"))

while True:
    print("\n1. Show Books | 2. Exit")
    choice = input("Select: ")
    if choice == "1":
        my_library.show_books()
    elif choice == "2":
        break
